@app.route('/chat', methods=['POST'])
def chat():
    try:
        r = requests.post(
            f"{OLLAMA_URL}/api/chat",
            json=request.json,
            stream=True,
            timeout=120
        )

        def generate():
            for chunk in r.iter_content(chunk_size=1024):
                if chunk:
                    yield chunk

        return Response(generate(), content_type="text/plain")

    except Exception as e:
        print("ERROR:", e)
        return jsonify({"error": str(e)}), 500
